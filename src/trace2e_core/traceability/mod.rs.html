1//! Traceability module.
2//!
3//! This module provides a complete traceability solution for distributed systems,
4//! enabling comprehensive data provenance tracking, policy enforcement, and compliance
5//! monitoring across process boundaries and network connections.
6//!
7//! ## Core Architecture
8//!
9//! The traceability system is built around three main API layers:
10//!
11//! ### Process-to-Middleware (P2M) API
12//! Primary interface for application processes to register resources and request
13//! I/O operations with traceability guarantees. Handles resource enrollment,
14//! authorization requests, and completion reporting.
15//!
16//! ### Middleware-to-Middleware (M2M) API  
17//! Enables communication between distributed middleware instances for cross-node
18//! policy evaluation, flow coordination, and provenance synchronization.
19//!
20//! ### Operator-to-Middleware (O2M) API
21//! Administrative interface for policy management, compliance configuration,
22//! and provenance analysis by external operators and organizations.
23//!
24//! ## Service Components
25//!
26//! ### Core Services
27//! - **Sequencer**: Manages flow ordering and resource reservations to prevent race conditions
28//! - **Provenance**: Tracks data provenance across operations
29//! - **Compliance**: Enforces organizational policies and regulatory requirements
30//!
31//! ### Infrastructure Services
32//! - **Validation**: Validates incoming requests and resource accessibility
33//! - **Naming**: Provides unified resource identification and naming conventions
34//! - **Error Handling**: Comprehensive error types and handling for operational monitoring
35//!
36//! ## Default Service Stacks
37//!
38//! Pre-configured service combinations are available as type aliases for common deployment patterns:
39//! - `M2mApiDefaultStack`: Standard M2M service with default component configuration
40//! - `P2mApiDefaultStack<M>`: Standard P2M service parameterized by M2M client type
41//! - `O2mApiDefaultStack`: Standard O2M service with default component configuration
42//!
43//! ## Initialization Helpers
44//!
45//! The module provides helper functions for initializing complete middleware stacks:
46//! - `init_middleware()`: Initialize production middleware with specified M2M client
47//! - `init_middleware_with_enrolled_resources()`: Initialize with pre-enrolled test resources

57/// Standard M2M API service stack with default component configuration.
58///
59/// Combines sequencer, provenance, and compliance services using the default
60/// waiting queue mechanism for flow coordination. Suitable for most production
61/// deployments requiring standard M2M functionality.
62pub type M2mApiDefaultStack = m2m::M2mApiService<
63    core::sequencer::WaitingQueueService<core::sequencer::SequencerService>,
64    core::provenance::ProvenanceService,
65    core::compliance::ComplianceService,
66>;

68/// Standard P2M API service stack parameterized by M2M client type.
69///
70/// Combines sequencer, provenance, and compliance services with a configurable
71/// M2M client for distributed coordination. The generic parameter `M` allows
72/// different M2M transport implementations to be used.
73pub type P2mApiDefaultStack<M> = p2m::P2mApiService<
74    core::sequencer::WaitingQueueService<core::sequencer::SequencerService>,
75    core::provenance::ProvenanceService,
76    core::compliance::ComplianceService,
77    M,
78>;

80/// Standard O2M API service stack with default component configuration.
81///
82/// Combines provenance and compliance services for administrative operations.
83/// Does not include sequencer as O2M operations are typically read-heavy
84/// and don't require flow coordination.
85pub type O2mApiDefaultStack =
86    o2m::O2mApiService<core::provenance::ProvenanceService, core::compliance::ComplianceService>;

88/// Initialize a complete middleware stack for production deployment.
89///
90/// Creates a fully configured middleware stack with all three API services
91/// (M2M, P2M, O2M) using default component configurations. This is the
92/// standard initialization method for production deployments.
93///
94/// # Arguments
95/// * `node_id` - Unique identifier for this middleware node in the distributed system
96/// * `max_retries` - Maximum retry attempts for the waiting queue (None for unlimited)
97/// * `m2m_client` - Client service for M2M communication with remote middleware
98/// * `enable_resource_validation` - Whether to enable resource validation for P2M requests
99///
100/// # Returns
101/// A tuple containing (M2M service, P2M service, O2M service) ready for use
102///
103/// # Type Parameters
104/// * `M` - M2M client type that implements the required service traits
105pub fn init_middleware<M>(
106    node_id: String,
107    max_retries: Option<u32>,
108    m2m_client: M,
109    enable_resource_validation: bool,
110) -> (M2mApiDefaultStack, P2mApiDefaultStack<M>, O2mApiDefaultStack)
111where
112    M: tower::Service<
113            api::M2mRequest,
114            Response = api::M2mResponse,
115            Error = error::TraceabilityError,
116        > + Clone
117        + Send
118        + 'static,
119    M::Future: Send,
120{
121    init_middleware_with_enrolled_resources(
122        node_id,
123        max_retries,
124        m2m_client,
125        enable_resource_validation,
126        0,
127        0,
128        0,
129    )
130}

132/// Initialize a middleware stack with pre-enrolled resources for testing.
133///
134/// Creates a middleware stack identical to `init_middleware()` but with
135/// pre-enrolled mock resources. This is useful for testing, benchmarking,
136/// and simulation scenarios where actual process interactions are not needed.
137///
138/// # Arguments
139/// * `node_id` - Unique identifier for this middleware node
140/// * `max_retries` - Maximum retry attempts for the waiting queue
141/// * `m2m_client` - Client service for M2M communication
142/// * `enable_resource_validation` - Whether to enable resource validation for P2M requests
143/// * `process_count` - Number of mock processes to pre-enroll
144/// * `per_process_file_count` - Number of files to enroll per process
145/// * `per_process_stream_count` - Number of streams to enroll per process
146///
147/// # Returns
148/// A tuple containing (M2M service, P2M service, O2M service) with pre-enrolled resources
149///
150/// # Warning
151/// Should only be used for testing purposes. Production deployments should use
152/// `init_middleware()` and rely on actual process enrollment.
153pub fn init_middleware_with_enrolled_resources<M>(
154    node_id: String,
155    max_retries: Option<u32>,
156    m2m_client: M,
157    enable_resource_validation: bool,
158    _process_count: u32,
159    _per_process_file_count: u32,
160    _per_process_stream_count: u32,
161) -> (M2mApiDefaultStack, P2mApiDefaultStack<M>, O2mApiDefaultStack)
162where
163    M: tower::Service<
164            api::M2mRequest,
165            Response = api::M2mResponse,
166            Error = error::TraceabilityError,
167        > + Clone
168        + Send
169        + 'static,
170    M::Future: Send,
171{
172    let sequencer = tower::ServiceBuilder::new()
173        .layer(tower::layer::layer_fn(|inner| {
174            core::sequencer::WaitingQueueService::new(inner, max_retries)
175        }))
176        .service(core::sequencer::SequencerService::default());
177    let provenance = core::provenance::ProvenanceService::new(node_id);
178    let compliance = core::compliance::ComplianceService::default();
179
180    let m2m_service: M2mApiDefaultStack =
181        m2m::M2mApiService::new(sequencer.clone(), provenance.clone(), compliance.clone());
182
183    #[cfg(test)]
184    let p2m_service: P2mApiDefaultStack<M> =
185        p2m::P2mApiService::new(sequencer, provenance.clone(), compliance.clone(), m2m_client)
186            .with_resource_validation(enable_resource_validation)
187            .with_enrolled_resources(
188                _process_count,
189                _per_process_file_count,
190                _per_process_stream_count,
191            );
192    #[cfg(not(test))]
193    let p2m_service: P2mApiDefaultStack<M> =
194        p2m::P2mApiService::new(sequencer, provenance.clone(), compliance.clone(), m2m_client)
195            .with_resource_validation(enable_resource_validation);
196
197    let o2m_service: O2mApiDefaultStack = o2m::O2mApiService::new(provenance, compliance);
198
199    (m2m_service, p2m_service, o2m_service)
200}
