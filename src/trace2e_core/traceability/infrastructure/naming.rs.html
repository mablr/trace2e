1//! Resource naming and identification system.
2//!
3//! This module defines the core resource identification and naming conventions used throughout
4//! the trace2e traceability system. It provides a unified way to represent and identify
5//! different types of computational resources including files, network streams, and processes.
6//!
7//! ## Resource Hierarchy
8//!
9//! **File Descriptors (Fd)**: Represent operating system file descriptors that can point to
10//! either files in the filesystem or network streams (sockets).
11//!
12//! **Files**: Filesystem resources identified by their path, supporting both absolute and
13//! relative path specifications.
14//!
15//! **Streams**: Network communication channels defined by local and peer socket addresses,
16//! supporting TCP connections, Unix domain sockets, and other network protocols.
17//!
18//! **Processes**: Running system processes identified by PID with additional metadata including
19//! start time and executable path for precise identification across process reuse.
20//!
21//! ## Resource Construction
22//!
23//! Resources can be constructed using dedicated factory methods that handle system queries
24//! and validation, or using mock variants for testing purposes.

34/// Represents a file resource in the filesystem.
35///
36/// Files are identified by their path, which can be absolute or relative.
37/// The path is stored as provided without normalization to preserve
38/// the original specification for audit and debugging purposes.
39#[derive(Debug, Clone, Eq, PartialEq, Hash)]
40pub struct File {
41    /// Filesystem path to the file, as specified by the application
42    pub path: String,
43}
44
45/// Represents a network stream or socket connection.
46///
47/// Streams are bidirectional communication channels between two endpoints,
48/// typically used for TCP connections, Unix domain sockets, or other
49/// network protocols. Both endpoints must be specified to enable proper
50/// flow tracking and policy enforcement.
51#[derive(Debug, Clone, Eq, PartialEq, Hash)]
52pub struct Stream {
53    /// Local socket address (e.g., "127.0.0.1:8080")
54    pub local_socket: String,
55    /// Remote peer socket address (e.g., "192.168.1.100:9000")
56    pub peer_socket: String,
57}
58
59/// Represents a file descriptor that can point to either a file or a stream.
60///
61/// File descriptors are the operating system's handle for I/O operations.
62/// This enum distinguishes between filesystem-based I/O (files) and
63/// network-based I/O (streams) while maintaining a unified interface.
64#[derive(Debug, Clone, Eq, PartialEq, Hash)]
65pub enum Fd {
66    /// File descriptor pointing to a filesystem file
67    File(File),
68    /// File descriptor pointing to a network stream or socket
69    Stream(Stream),
70}
71
72/// Represents a running system process with identifying metadata.
73///
74/// Processes are identified not only by their PID (which can be reused)
75/// but also by their start time and executable path to ensure precise
76/// identification across the system lifecycle.
77#[derive(Debug, Clone, Eq, PartialEq, Hash)]
78pub struct Process {
79    /// Process identifier assigned by the operating system
80    pub pid: i32,
81    /// Process start time in seconds since epoch for uniqueness
82    pub starttime: u64,
83    /// Path to the executable that created this process
84    pub exe_path: String,
85}
86
87/// Unified resource identifier for all trackable entities in the system.
88///
89/// Resources represent any entity that can participate in data flows within
90/// the traceability system. This includes file descriptors (which may point
91/// to files or streams), processes, or null resources for uninitialized states.
92#[derive(Debug, Clone, Eq, PartialEq, Hash, Default)]
93pub enum Resource {
94    /// File descriptor resource (file or stream)
95    Fd(Fd),
96    /// Process resource  
97    Process(Process),
98    /// Null or uninitialized resource
99    #[default]
100    None,
101}
102
103impl Resource {
104    /// Creates a new file resource with the specified filesystem path.
105    ///
106    /// The path is stored as provided without validation or normalization.
107    /// Applications should provide valid paths to ensure proper resource tracking.
108    pub fn new_file(path: String) -> Self {
109        Self::Fd(Fd::File(File { path }))
110    }
111
112    /// Creates a new stream resource with the specified socket addresses.
113    ///
114    /// Both local and peer socket addresses should be valid network addresses
115    /// in the format appropriate for the protocol (e.g., "IP:port" for TCP).
116    pub fn new_stream(local_socket: String, peer_socket: String) -> Self {
117        Self::Fd(Fd::Stream(Stream { local_socket, peer_socket }))
118    }
119
120    /// Creates a new process resource by querying the system for process information.
121    ///
122    /// Attempts to retrieve the process start time and executable path from the
123    /// system process table. If the process is not found, creates a process
124    /// resource with default values for the metadata fields.
125    pub fn new_process(pid: i32) -> Self {
126        let mut system = System::new();
127        system.refresh_all();
128        if let Some(process) = system.process(Pid::from(pid as usize)) {
129            let starttime = process.start_time();
130            let exe_path = if let Some(exe) = process.exe() {
131                exe.to_string_lossy().to_string()
132            } else {
133                String::new()
134            };
135            Self::Process(Process { pid, starttime, exe_path })
136        } else {
137            Self::Process(Process { pid, starttime: 0, exe_path: String::new() })
138        }
139    }
140
141    /// Creates a mock process resource for testing purposes.
142    ///
143    /// Creates a process resource with the specified PID but default values
144    /// for start time and executable path. Should only be used in test
145    /// environments where system process queries are not needed.
146    pub fn new_process_mock(pid: i32) -> Self {
147        Self::Process(Process { pid, starttime: 0, exe_path: String::new() })
148    }
149
150    /// Checks if this resource represents a filesystem file.
151    ///
152    /// Returns true if the resource is a file descriptor pointing to a file,
153    /// false for streams, processes, or null resources.
154    pub fn is_file(&self) -> bool {
155        matches!(self, Resource::Fd(Fd::File(_)))
156    }
157
158    /// Checks if this resource represents a network stream.
159    ///
160    /// Returns true if the resource is a file descriptor pointing to a stream,
161    /// false for files, processes, or null resources.
162    pub fn is_stream(&self) -> bool {
163        matches!(self, Resource::Fd(Fd::Stream(_)))
164    }
165
166    /// Converts this resource into a localized resource given the specified localization.
167    ///
168    /// It attempts to convert the resource into a localized stream resource, which infers
169    /// the peer node from the stream's peer socket.
170    /// Otherwise, it creates a localized resource with the specified localization..
171    pub fn into_localized(self, localization: String) -> LocalizedResource {
172        if let Some(localized_stream) = self.try_into_localized_peer_stream() {
173            localized_stream
174        } else {
175            LocalizedResource::new(localization, self)
176        }
177    }
178
179    /// Returns the peer stream resource if this is a stream resource.
180    ///
181    /// For stream resources, returns a new resource with the local and peer
182    /// socket addresses swapped. This is useful for tracking bidirectional
183    /// flows. Returns None for non-stream resources.
184    pub fn try_into_localized_peer_stream(&self) -> Option<LocalizedResource> {
185        if let Resource::Fd(Fd::Stream(stream)) = self
186            && let Ok(peer_socket) = stream.peer_socket.parse::<SocketAddr>()
187        {
188            Some(LocalizedResource::new(
189                peer_socket.ip().to_string(),
190                Resource::new_stream(stream.peer_socket.to_owned(), stream.local_socket.to_owned()),
191            ))
192        } else {
193            None
194        }
195    }
196
197    /// Checks if this resource represents a system process.
198    ///
199    /// Returns true if the resource is a process, false for file descriptors
200    /// or null resources.
201    pub fn is_process(&self) -> bool {
202        matches!(self, Resource::Process(_))
203    }
204}
205
206impl Display for Resource {
207    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
208        match self {
209            Resource::Fd(Fd::File(file)) => write!(f, "file:://{}", file.path),
210            Resource::Fd(Fd::Stream(stream)) => {
211                write!(f, "stream:://{}::{}", stream.local_socket, stream.peer_socket)
212            }
213            Resource::Process(process) => {
214                write!(f, "process:://{}::{}::{}", process.pid, process.starttime, process.exe_path)
215            }
216            Resource::None => write!(f, "None"),
217        }
218    }
219}

221/// Unified resource identifier for all trackable entities in the system.
222///
223/// Localized resources are resources that are associated with a specific node in a distributed system.
224/// They are used to identify resources that are local to a specific node and are used to track
225/// resources that are local to a specific node.
226#[derive(Debug, Clone, Eq, PartialEq, Hash, Default)]
227pub struct LocalizedResource {
228    /// Node identifier for this localized resource
229    node_id: String,
230    /// Resource for this localized resource
231    resource: Resource,
232}
233
234impl LocalizedResource {
235    /// Creates a new localized resource with the specified node identifier and resource.
236    pub fn new(node_id: String, resource: Resource) -> Self {
237        Self { node_id, resource }
238    }
239
240    /// Returns the node identifier for this localized resource.
241    pub fn node_id(&self) -> &String {
242        &self.node_id
243    }
244
245    /// Returns the resource for this localized resource.
246    pub fn resource(&self) -> &Resource {
247        &self.resource
248    }
249}
250
251impl Display for LocalizedResource {
252    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
253        write!(f, "{}@{}", self.resource, self.node_id)
254    }
255}

257/// Wrapper type for Display implementation of Option<Resource>
258///
259/// This wrapper is needed because Rust's orphan rule prevents implementing
260/// Display directly for Option<Resource> when Option is from the standard library.
261#[derive(Clone, Debug)]
262pub enum DisplayableResource<T> {
263    Option(Option<T>),
264    HashSet(HashSet<T>),
265    Vec(Vec<T>),
266}
267
268impl Display for DisplayableResource<Resource> {
269    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
270        match self {
271            DisplayableResource::Option(Some(resource)) => write!(f, "{}", resource),
272            DisplayableResource::Option(None) => write!(f, "None"),
273            DisplayableResource::HashSet(resources) => write!(
274                f,
275                "[{}]",
276                resources.iter().map(|r| r.to_string()).collect::<Vec<String>>().join(", ")
277            ),
278            DisplayableResource::Vec(resources) => write!(
279                f,
280                "[{}]",
281                resources.iter().map(|r| r.to_string()).collect::<Vec<String>>().join(", ")
282            ),
283        }
284    }
285}

342/// Trait for services that have a node identifier in distributed systems.
343///
344/// This trait is implemented by services that participate in distributed
345/// traceability operations and need to identify themselves to remote peers.
346/// The node ID is typically used in provenance records and logging to
347/// track which middleware instance processed specific operations.
348pub trait NodeId {
349    /// Returns the unique identifier for this node in the distributed system.
350    ///
351    /// Node IDs should be unique across the distributed deployment and
352    /// persistent across service restarts to maintain consistent provenance records.
353    fn node_id(&self) -> String;
354}

360    #[test]
361    fn test_resource_display() {
362        let file = LocalizedResource::new(
363            "127.0.0.1".to_string(),
364            Resource::new_file("/tmp/test.txt".to_string()),
365        );
366        let stream = LocalizedResource::new(
367            "127.0.0.1".to_string(),
368            Resource::new_stream("127.0.0.1:8080".to_string(), "127.0.0.1:8081".to_string()),
369        );
370        let process_mock =
371            LocalizedResource::new("127.0.0.1".to_string(), Resource::new_process_mock(1234));
372        let none = LocalizedResource::new(Default::default(), Resource::None);
373
374        assert_eq!(file.to_string(), "file::///tmp/test.txt@127.0.0.1");
375        assert_eq!(stream.to_string(), "stream:://127.0.0.1:8080::127.0.0.1:8081@127.0.0.1");
376        assert_eq!(process_mock.to_string(), "process:://1234::0::@127.0.0.1");
377        assert_eq!(none.to_string(), "None@");
378    }

381    fn test_displayable_resource() {

391            LocalizedResource::new("127.0.0.1".to_string(), Resource::new_process_mock(1234));

407        assert_eq!(DisplayableResource::from(Some(none.clone())).to_string(), "None@");
408
409        assert_eq!(
