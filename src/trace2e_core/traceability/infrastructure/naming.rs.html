1//! Resource naming and identification system.
2//!
3//! This module defines the core resource identification and naming conventions used throughout
4//! the trace2e traceability system. It provides a unified way to represent and identify
5//! different types of computational resources including files, network streams, and processes.
6//!
7//! ## Resource Hierarchy
8//!
9//! **File Descriptors (Fd)**: Represent operating system file descriptors that can point to
10//! either files in the filesystem or network streams (sockets).
11//!
12//! **Files**: Filesystem resources identified by their path, supporting both absolute and
13//! relative path specifications.
14//!
15//! **Streams**: Network communication channels defined by local and peer socket addresses,
16//! supporting TCP connections, Unix domain sockets, and other network protocols.
17//!
18//! **Processes**: Running system processes identified by PID with additional metadata including
19//! start time and executable path for precise identification across process reuse.
20//!
21//! ## Resource Construction
22//!
23//! Resources can be constructed using dedicated factory methods that handle system queries
24//! and validation, or using mock variants for testing purposes.
25

27    fmt::{Debug, Display},

31use sysinfo::{Pid, System};

33/// Represents a file resource in the filesystem.
34///
35/// Files are identified by their path, which can be absolute or relative.
36/// The path is stored as provided without normalization to preserve
37/// the original specification for audit and debugging purposes.
38#[derive(Debug, Clone, Eq, PartialEq, Hash)]
39pub struct File {
40    /// Filesystem path to the file, as specified by the application
41    pub path: String,
42}
43
44/// Represents a network stream or socket connection.
45///
46/// Streams are bidirectional communication channels between two endpoints,
47/// typically used for TCP connections, Unix domain sockets, or other
48/// network protocols. Both endpoints must be specified to enable proper
49/// flow tracking and policy enforcement.
50#[derive(Debug, Clone, Eq, PartialEq, Hash)]
51pub struct Stream {
52    /// Local socket address (e.g., "127.0.0.1:8080")
53    pub local_socket: String,
54    /// Remote peer socket address (e.g., "192.168.1.100:9000")
55    pub peer_socket: String,
56}
57
58/// Represents a file descriptor that can point to either a file or a stream.
59///
60/// File descriptors are the operating system's handle for I/O operations.
61/// This enum distinguishes between filesystem-based I/O (files) and
62/// network-based I/O (streams) while maintaining a unified interface.
63#[derive(Debug, Clone, Eq, PartialEq, Hash)]
64pub enum Fd {
65    /// File descriptor pointing to a filesystem file
66    File(File),
67    /// File descriptor pointing to a network stream or socket
68    Stream(Stream),
69}
70
71/// Represents a running system process with identifying metadata.
72///
73/// Processes are identified not only by their PID (which can be reused)
74/// but also by their start time and executable path to ensure precise
75/// identification across the system lifecycle.
76#[derive(Debug, Clone, Eq, PartialEq, Hash)]
77pub struct Process {
78    /// Process identifier assigned by the operating system
79    pub pid: i32,
80    /// Process start time in seconds since epoch for uniqueness
81    pub starttime: u64,
82    /// Path to the executable that created this process
83    pub exe_path: String,
84}
85
86/// Unified resource identifier for all trackable entities in the system.
87///
88/// Resources represent any entity that can participate in data flows within
89/// the traceability system. This includes file descriptors (which may point
90/// to files or streams), processes, or null resources for uninitialized states.
91#[derive(Debug, Clone, Eq, PartialEq, Hash, Default)]
92pub enum Resource {
93    /// File descriptor resource (file or stream)
94    Fd(Fd),
95    /// Process resource  
96    Process(Process),
97    /// Null or uninitialized resource
98    #[default]
99    None,
100}
101
102impl Resource {
103    /// Creates a new file resource with the specified filesystem path.
104    ///
105    /// The path is stored as provided without validation or normalization.
106    /// Applications should provide valid paths to ensure proper resource tracking.
107    pub fn new_file(path: String) -> Self {
108        Self::Fd(Fd::File(File { path }))
109    }
110
111    /// Creates a new stream resource with the specified socket addresses.
112    ///
113    /// Both local and peer socket addresses should be valid network addresses
114    /// in the format appropriate for the protocol (e.g., "IP:port" for TCP).
115    pub fn new_stream(local_socket: String, peer_socket: String) -> Self {
116        Self::Fd(Fd::Stream(Stream { local_socket, peer_socket }))
117    }
118
119    /// Creates a new process resource by querying the system for process information.
120    ///
121    /// Attempts to retrieve the process start time and executable path from the
122    /// system process table. If the process is not found, creates a process
123    /// resource with default values for the metadata fields.
124    pub fn new_process(pid: i32) -> Self {
125        let mut system = System::new();
126        system.refresh_all();
127        if let Some(process) = system.process(Pid::from(pid as usize)) {
128            let starttime = process.start_time();
129            let exe_path = if let Some(exe) = process.exe() {
130                exe.to_string_lossy().to_string()
131            } else {
132                String::new()
133            };
134            Self::Process(Process { pid, starttime, exe_path })
135        } else {
136            Self::Process(Process { pid, starttime: 0, exe_path: String::new() })
137        }
138    }
139
140    /// Creates a mock process resource for testing purposes.
141    ///
142    /// Creates a process resource with the specified PID but default values
143    /// for start time and executable path. Should only be used in test
144    /// environments where system process queries are not needed.
145    pub fn new_process_mock(pid: i32) -> Self {
146        Self::Process(Process { pid, starttime: 0, exe_path: String::new() })
147    }
148
149    /// Checks if this resource represents a filesystem file.
150    ///
151    /// Returns true if the resource is a file descriptor pointing to a file,
152    /// false for streams, processes, or null resources.
153    pub fn is_file(&self) -> bool {
154        matches!(self, Resource::Fd(Fd::File(_)))
155    }
156
157    /// Checks if this resource represents a network stream.
158    ///
159    /// Returns true if the resource is a file descriptor pointing to a stream,
160    /// false for files, processes, or null resources.
161    pub fn is_stream(&self) -> bool {
162        matches!(self, Resource::Fd(Fd::Stream(_)))
163    }
164
165    /// Converts this resource into a localized resource given the specified localization.
166    ///
167    /// It attempts to convert the resource into a localized stream resource, which infers
168    /// the peer node from the stream's peer socket.
169    /// Otherwise, it creates a localized resource with the specified localization..
170    pub fn into_localized(self, localization: String) -> LocalizedResource {
171        if let Some(localized_stream) = self.try_into_localized_peer_stream() {
172            localized_stream
173        } else {
174            LocalizedResource::new(localization, self)
175        }
176    }
177
178    /// Returns the peer stream resource if this is a stream resource.
179    ///
180    /// For stream resources, returns a new resource with the local and peer
181    /// socket addresses swapped. This is useful for tracking bidirectional
182    /// flows. Returns None for non-stream resources.
183    pub fn try_into_localized_peer_stream(&self) -> Option<LocalizedResource> {
184        if let Resource::Fd(Fd::Stream(stream)) = self
185            && let Ok(peer_socket) = stream.peer_socket.parse::<SocketAddr>()
186        {
187            Some(LocalizedResource::new(
188                peer_socket.ip().to_string(),
189                Resource::new_stream(stream.peer_socket.to_owned(), stream.local_socket.to_owned()),
190            ))
191        } else {
192            None
193        }
194    }
195
196    /// Checks if this resource represents a system process.
197    ///
198    /// Returns true if the resource is a process, false for file descriptors
199    /// or null resources.
200    pub fn is_process(&self) -> bool {
201        matches!(self, Resource::Process(_))
202    }
203}
204
205impl Display for Resource {
206    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
207        match self {
208            Resource::Fd(Fd::File(file)) => write!(f, "file:://{}", file.path),
209            Resource::Fd(Fd::Stream(stream)) => {
210                write!(f, "stream:://{}::{}", stream.local_socket, stream.peer_socket)
211            }
212            Resource::Process(process) => {
213                write!(f, "process:://{}::{}::{}", process.pid, process.starttime, process.exe_path)
214            }
215            Resource::None => write!(f, "None"),
216        }
217    }
218}

220/// Unified resource identifier for all trackable entities in the system.
221///
222/// Localized resources are resources that are associated with a specific node in a distributed system.
223/// They are used to identify resources that are local to a specific node and are used to track
224/// resources that are local to a specific node.
225#[derive(Debug, Clone, Eq, PartialEq, Hash, Default)]
226pub struct LocalizedResource {
227    /// Node identifier for this localized resource
228    node_id: String,
229    /// Resource for this localized resource
230    resource: Resource,
231}
232
233impl LocalizedResource {
234    /// Creates a new localized resource with the specified node identifier and resource.
235    pub fn new(node_id: String, resource: Resource) -> Self {
236        Self { node_id, resource }
237    }
238
239    /// Returns the node identifier for this localized resource.
240    pub fn node_id(&self) -> &String {
241        &self.node_id
242    }
243
244    /// Returns the resource for this localized resource.
245    pub fn resource(&self) -> &Resource {
246        &self.resource
247    }
248}
249
250impl Display for LocalizedResource {
251    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
252        write!(f, "{}@{}", self.resource, self.node_id)
253    }
254}

256/// Trait for services that have a node identifier in distributed systems.
257///
258/// This trait is implemented by services that participate in distributed
259/// traceability operations and need to identify themselves to remote peers.
260/// The node ID is typically used in provenance records and logging to
261/// track which middleware instance processed specific operations.
262pub trait NodeId {
263    /// Returns the unique identifier for this node in the distributed system.
264    ///
265    /// Node IDs should be unique across the distributed deployment and
266    /// persistent across service restarts to maintain consistent provenance records.
267    fn node_id(&self) -> String;
268}

270#[cfg(test)]
271mod tests {
272    use super::*;
273
274    #[test]
275    fn test_resource_display() {
276        let file = LocalizedResource::new(
277            "127.0.0.1".to_string(),
278            Resource::new_file("/tmp/test.txt".to_string()),
279        );
280        let stream = LocalizedResource::new(
281            "127.0.0.1".to_string(),
282            Resource::new_stream("127.0.0.1:8080".to_string(), "127.0.0.1:8081".to_string()),
283        );
284        let process_mock =
285            LocalizedResource::new("127.0.0.1".to_string(), Resource::new_process_mock(1234));
286        let none = LocalizedResource::new(Default::default(), Resource::None);
287
288        assert_eq!(file.to_string(), "file::///tmp/test.txt@127.0.0.1");
289        assert_eq!(stream.to_string(), "stream:://127.0.0.1:8080::127.0.0.1:8081@127.0.0.1");
290        assert_eq!(process_mock.to_string(), "process:://1234::0::@127.0.0.1");
291        assert_eq!(none.to_string(), "None@");
292    }
